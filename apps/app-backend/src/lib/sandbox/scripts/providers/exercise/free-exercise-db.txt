const GITHUB_EXERCISES_URL =
  "https://raw.githubusercontent.com/yuhonas/free-exercise-db/main/dist/exercises.json";

const IMAGES_PREFIX_URL =
  "https://raw.githubusercontent.com/yuhonas/free-exercise-db/main/exercises";

const EXERCISES_CACHE_KEY = "exercise.free-exercise-db.all";
const EXERCISES_CACHE_TTL_SECONDS = 86400;

const categoryToLot = (category) => {
  const lower = category.toLowerCase();
  if (lower === "cardio") return "distance_and_duration";
  if (lower === "stretching" || lower === "plyometrics") return "duration";
  if (
    lower === "strongman" ||
    lower === "olympic weightlifting" ||
    lower === "strength" ||
    lower === "powerlifting"
  )
    return "reps_and_weight";
  return null;
};

const normalizeSlug = (value) => {
  if (!value) return null;
  return value.toLowerCase().replace(/\s+/g, "_");
};

const normalizeEquipment = (value) => {
  if (!value) return null;
  const lower = value.toLowerCase();
  if (lower === "e-z curl bar") return "ez_curl_bar";
  return normalizeSlug(value);
};

const normalizeMuscles = (arr) => {
  if (!Array.isArray(arr)) return [];
  return arr.flatMap((m) => {
    const n = normalizeSlug(m);
    return n ? [n] : [];
  });
};

async function fetchAllExercises() {
  const cached = await getCachedValue(EXERCISES_CACHE_KEY);
  if (cached?.success && typeof cached.data === "string") {
    try {
      return JSON.parse(cached.data);
    } catch {
      // cache miss or stale, fall through to re-fetch
    }
  }

  const response = await httpCall("GET", GITHUB_EXERCISES_URL, {});
  if (!response?.success) {
    throw new Error(response?.error ?? "Failed to fetch exercises from free-exercise-db");
  }

  let exercises;
  try {
    exercises = JSON.parse(response.data.body);
  } catch {
    throw new Error("Free Exercise DB returned invalid JSON");
  }

  await setCachedValue(
    EXERCISES_CACHE_KEY,
    JSON.stringify(exercises),
    EXERCISES_CACHE_TTL_SECONDS,
  );

  return exercises;
}

driver("search", async function (context) {
  const { z } = await import("npm:zod");

  const { query, page, pageSize } = z
    .object({
      query: z.string().trim().min(1, "query is required"),
      page: z.coerce.number().min(1).transform(Math.floor).catch(1),
      pageSize: z.coerce.number().min(1).max(100).transform(Math.floor).catch(20),
    })
    .parse(context ?? {});

  const allExercises = await fetchAllExercises();

  const queryLower = query.toLowerCase();
  const filtered = allExercises.filter((ex) => {
    const name = typeof ex.name === "string" ? ex.name.toLowerCase() : "";
    return name.includes(queryLower);
  });

  const totalItems = filtered.length;
  const start = (page - 1) * pageSize;
  const pageItems = filtered.slice(start, start + pageSize);

  const items = pageItems.flatMap((ex) => {
    const name = typeof ex.name === "string" ? ex.name : "";
    if (!name) return [];

    const lot = categoryToLot(typeof ex.category === "string" ? ex.category : "");
    if (!lot) return [];

    const level = normalizeSlug(typeof ex.level === "string" ? ex.level : null);
    const equipment = normalizeEquipment(
      typeof ex.equipment === "string" ? ex.equipment : null,
    );
    const firstImagePath = Array.isArray(ex.images) ? ex.images[0] : null;
    const imageUrl = firstImagePath ? `${IMAGES_PREFIX_URL}/${firstImagePath}` : null;

    return [
      {
        externalId: name,
        titleProperty: { kind: "text", value: name },
        calloutProperty: level
          ? { kind: "text", value: level }
          : { kind: "null", value: null },
        primarySubtitleProperty: { kind: "text", value: lot },
        secondarySubtitleProperty: equipment
          ? { kind: "text", value: equipment }
          : { kind: "null", value: null },
        imageProperty: imageUrl
          ? { kind: "image", value: { kind: "remote", url: imageUrl } }
          : { kind: "null", value: null },
      },
    ];
  });

  return {
    items,
    details: {
      totalItems,
      nextPage: page * pageSize < totalItems ? page + 1 : null,
    },
  };
});

driver("details", async function (context) {
  const { z } = await import("npm:zod");

  const { externalId } = z
    .object({
      externalId: z.string().trim().min(1, "externalId is required"),
    })
    .parse(context ?? {});

  const allExercises = await fetchAllExercises();
  const ex = allExercises.find((e) => e.name === externalId);

  if (!ex) {
    throw new Error(`Exercise not found: ${externalId}`);
  }

  const lot = categoryToLot(typeof ex.category === "string" ? ex.category : "");
  if (!lot) {
    throw new Error(
      `Unrecognized category "${ex.category}" for exercise: ${externalId}`,
    );
  }

  const rawImages = Array.isArray(ex.images) ? ex.images : [];
  const images = rawImages.map((path) => ({
    kind: "remote",
    url: `${IMAGES_PREFIX_URL}/${path}`,
  }));

  const primaryMuscles = normalizeMuscles(ex.primaryMuscles);
  const secondaryMuscles = normalizeMuscles(ex.secondaryMuscles);
  const muscles = [...new Set([...primaryMuscles, ...secondaryMuscles])];

  return {
    name: ex.name,
    properties: {
      lot,
      images,
      muscles,
      source: "github",
      force: normalizeSlug(typeof ex.force === "string" ? ex.force : null),
      level: normalizeSlug(typeof ex.level === "string" ? ex.level : null),
      mechanic: normalizeSlug(typeof ex.mechanic === "string" ? ex.mechanic : null),
      equipment: normalizeEquipment(
        typeof ex.equipment === "string" ? ex.equipment : null,
      ),
      instructions: Array.isArray(ex.instructions) ? ex.instructions : [],
    },
  };
});
