function parseJsonResponse(responseBody) {
  try {
    return JSON.parse(responseBody);
  } catch {
    throw new Error("Hardcover returned invalid JSON");
  }
}

const HARDCOVER_GQL_URL = "https://api.hardcover.app/v1/graphql";

async function getHardcoverApiKey() {
  const resp = await getAppConfigValue("books.hardcover.apiKey");
  if (!resp?.success) {
    throw new Error(resp?.error ?? "Could not load Hardcover API key");
  }
  const apiKey = typeof resp.data === "string" ? resp.data.trim() : "";
  if (!apiKey) throw new Error("BOOKS_HARDCOVER_API_KEY is not configured");
  return apiKey;
}

async function hardcoverGql(query, apiKey) {
  const response = await httpCall("POST", HARDCOVER_GQL_URL, {
    body: JSON.stringify({ query }),
    headers: { Authorization: apiKey, "Content-Type": "application/json" },
  });
  if (!response?.success) {
    throw new Error(response?.error ?? "Hardcover GraphQL request failed");
  }
  return parseJsonResponse(response.data.body);
}

driver("search", async function (context) {
  const { z } = await import("npm:zod");

  const { query, page: currentPage, pageSize } = z
    .object({
      query: z.string().trim().min(1, "query is required"),
      page: z.coerce.number().min(1).transform(Math.floor).catch(1),
      pageSize: z.coerce.number().min(1).max(100).transform(Math.floor).catch(20),
    })
    .parse(context ?? {});

  const apiKey = await getHardcoverApiKey();

  const gqlQuery = `
query {
  search(
    page: ${currentPage},
    per_page: ${pageSize},
    query: "${query.replace(/"/g, '\\"')}",
    query_type: "series"
  ) {
    results
  }
}
`;
  const payload = await hardcoverGql(gqlQuery, apiKey);
  const resultsData =
    payload?.data?.search?.results &&
    typeof payload.data.search.results === "object"
      ? payload.data.search.results
      : null;

  if (!resultsData) throw new Error("Hardcover returned invalid response structure");

  const totalItems =
    typeof resultsData.found === "number" ? Math.max(0, Math.trunc(resultsData.found)) : 0;

  const hits = Array.isArray(resultsData.hits) ? resultsData.hits : [];
  const items = hits
    .map((hit) => {
      const doc = hit?.document;
      if (!doc || typeof doc !== "object") return null;
      const id =
        doc.id != null ? String(doc.id).trim() : null;
      if (!id) return null;
      const name =
        typeof doc.name === "string" && doc.name.trim() ? doc.name.trim() : null;
      if (!name) return null;
      const parts =
        typeof doc.books_count === "number" ? doc.books_count : null;
      const imageUrl =
        doc.image && typeof doc.image === "object" && typeof doc.image.url === "string"
          ? doc.image.url.trim() || null
          : null;
      return {
        externalId: id,
        calloutProperty: { kind: "null", value: null },
        titleProperty: { kind: "text", value: name },
        primarySubtitleProperty:
          parts === null
            ? { kind: "null", value: null }
            : { kind: "number", value: parts },
        secondarySubtitleProperty: { kind: "null", value: null },
        imageProperty:
          imageUrl === null
            ? { kind: "null", value: null }
            : { kind: "image", value: { type: "remote", url: imageUrl } },
      };
    })
    .filter((item) => item !== null);

  return {
    items,
    details: {
      totalItems,
      nextPage: currentPage * pageSize < totalItems ? currentPage + 1 : null,
    },
  };
});

driver("details", async function (context) {
  const { z } = await import("npm:zod");

  const { externalId } = z
    .object({ externalId: z.string().trim().min(1, "externalId is required") })
    .parse(context ?? {});

  const apiKey = await getHardcoverApiKey();

  const gqlQuery = `
{
  series_by_pk(id: ${externalId}) {
    id
    name
    slug
    books_count
    description
    book_series(
      order_by: { position: asc }
      where: {
        book: {
          book_status_id: { _eq: "1" }, compilation: { _eq: false },
          default_physical_edition: { language_id: { _eq: 1 } }
        }
      }
    ) {
      book { id title }
    }
  }
}
`;
  const payload = await hardcoverGql(gqlQuery, apiKey);
  const data = payload?.data?.series_by_pk;
  if (!data || typeof data !== "object") {
    throw new Error("Hardcover returned no series data");
  }

  const title =
    typeof data.name === "string" && data.name.trim() ? data.name.trim() : null;
  if (!title) throw new Error("Hardcover series is missing name");

  const description =
    typeof data.description === "string" && data.description.trim()
      ? data.description.trim()
      : null;

  const parts =
    typeof data.books_count === "number" ? data.books_count : null;

  const sourceUrl =
    typeof data.slug === "string" && data.slug.trim()
      ? `https://hardcover.app/series/${data.slug.trim()}`
      : null;

  return {
    name: title,
    properties: {
      parts,
      sourceUrl,
      images: [],
      description,
    },
  };
});
