function parseJsonResponse(responseBody) {
  try {
    return JSON.parse(responseBody);
  } catch {
    throw new Error("Audnex returned invalid JSON");
  }
}

const PAGE_SIZE = 20;

driver("search", async function(context) {
  const pageSize =
    Number.isFinite(Number(context?.pageSize)) &&
    Number(context.pageSize) > 0 &&
    Number(context.pageSize) <= 100
      ? Math.floor(Number(context.pageSize))
      : PAGE_SIZE;
  const currentPage = Number.isFinite(Number(context?.page))
    ? Math.max(1, Math.floor(Number(context.page)))
    : 1;
  const query = typeof context?.query === "string" ? context.query.trim() : "";

  if (!query) {
    throw new Error("query is required");
  }

  const params = new URLSearchParams({
    name: query,
    region: "us",
  });

  const response = await httpCall(
    "GET",
    `https://api.audnex.us/authors?${params.toString()}`,
  );

  if (!response?.success) {
    throw new Error(response?.error ?? "Audnex author search request failed");
  }

  const data = parseJsonResponse(response.data.body);
  const allItems = Array.isArray(data) ? data : [];

  const totalItems = allItems.length;

  const startIndex = (currentPage - 1) * pageSize;
  const endIndex = Math.min(startIndex + pageSize, totalItems);
  const pageItems = allItems.slice(startIndex, endIndex);

  const items = pageItems
    .map((author) => {
      const identifier =
        typeof author?.asin === "string" ? author.asin.trim() : "";
      if (!identifier) {
        return null;
      }

      const name = typeof author?.name === "string" ? author.name.trim() : "";

      return {
        identifier,
        badgeProperty: { kind: "null", value: null },
        titleProperty: { kind: "text", value: name },
        subtitleProperty: { kind: "null", value: null },
        imageProperty: { kind: "null", value: null },
      };
    })
    .filter((item) => item !== null);

  return {
    items,
    details: {
      totalItems,
      nextPage: endIndex < totalItems ? currentPage + 1 : null,
    },
  };
});

driver("details", async function(context) {
  const identifier =
    typeof context?.identifier === "string" ? context.identifier.trim() : "";
  if (!identifier) {
    throw new Error("identifier is required");
  }

  const params = new URLSearchParams({ region: "us" });

  const response = await httpCall(
    "GET",
    `https://api.audnex.us/authors/${identifier}?${params.toString()}`,
  );

  if (!response?.success) {
    throw new Error(response?.error ?? "Audnex author details request failed");
  }

  const data = parseJsonResponse(response.data.body);

  const name = typeof data?.name === "string" ? data.name.trim() : "";
  if (!name) {
    throw new Error("Audnex returned no author name");
  }

  const image =
    typeof data?.image === "string" && data.image.trim()
      ? data.image.trim()
      : null;

  const description =
    typeof data?.description === "string" ? data.description : null;

  const sourceUrl = `https://www.audible.com/author/${identifier}`;

  return {
    name,
    properties: {
      gender: null,
      website: null,
      birthDate: null,
      deathDate: null,
      birthPlace: null,
      sourceUrl,
      description,
      alternateNames: [],
      assets: { remoteImages: image ? [image] : [] },
    },
  };
});
