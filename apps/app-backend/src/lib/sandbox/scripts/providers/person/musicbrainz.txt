function parseJsonResponse(responseBody) {
  try {
    return JSON.parse(responseBody);
  } catch {
    throw new Error("MusicBrainz returned invalid JSON");
  }
}

function getString(value) {
  return typeof value === "string" ? value.trim() : "";
}

function getNullableString(value) {
  const s = getString(value);
  return s || null;
}

const MB_BASE = "https://musicbrainz.org/ws/2";
const MB_HEADERS = {
  Accept: "application/json",
};
const PAGE_SIZE = 20;

async function mbGet(path, params) {
  const search = new URLSearchParams({ fmt: "json", ...params });
  const url = `${MB_BASE}/${path}?${search.toString()}`;
  const response = await httpCall("GET", url, { headers: MB_HEADERS });
  if (!response?.success) {
    if (response?.data?.status === 404) {
      return null;
    }
    throw new Error(response?.error ?? `MusicBrainz request failed: ${path}`);
  }

  return parseJsonResponse(response.data.body);
}

function buildLuceneQuery(query, fields) {
  const escaped = query.replace(/([+\-!(){}[\]^"~*?:\\/])/g, "\\$1");
  return fields.map((f) => `${f}:(${escaped})`).join(" OR ");
}

function buildArtistDescription(artist) {
  const parts = [];

  const artistType = getNullableString(artist?.type);
  if (artistType) {
    parts.push(artistType);
  }

  const country = getNullableString(artist?.country);
  if (country) {
    parts.push(`Country: ${country}`);
  }

  const area = getNullableString(artist?.area?.name);
  if (area) {
    parts.push(`Area: ${area}`);
  }

  const lifeSpan = artist?.["life-span"];
  if (lifeSpan && typeof lifeSpan === "object") {
    const begin = getNullableString(lifeSpan.begin);
    const end = getNullableString(lifeSpan.end);
    if (begin || end) {
      parts.push(`Active: ${begin || "?"} - ${end || "?"}`);
    }
  }

  const disambiguation = getNullableString(artist?.disambiguation);
  if (disambiguation) {
    parts.push(disambiguation);
  }

  return parts.length > 0 ? parts.join(" - ") : null;
}

driver("search", async function (context) {
  const { z } = await import("npm:zod");
  const { query, page: currentPage, pageSize } = z.object({
    query: z.string().trim().min(1, "query is required"),
    page: z.coerce.number().min(1).transform(Math.floor).catch(1),
    pageSize: z.coerce.number().min(1).max(100).transform(Math.floor).catch(PAGE_SIZE),
  }).parse(context ?? {});

  const luceneQuery = buildLuceneQuery(query, ["artist", "alias"]);
  const data = await mbGet("artist", {
    query: luceneQuery,
    limit: String(pageSize),
    offset: String((currentPage - 1) * pageSize),
  });

  if (!data) {
    throw new Error("MusicBrainz artist search returned no data");
  }

  const totalItems =
    typeof data.count === "number" ? Math.max(0, data.count) : 0;
  const artists = Array.isArray(data.artists) ? data.artists : [];

  const items = artists
    .map((artist) => {
      const id = getString(artist?.id);
      if (!id) {
        return null;
      }

      const name = getString(artist?.name) || id;

      return {
        externalId: id,
        badgeProperty: { kind: "null", value: null },
        titleProperty: { kind: "text", value: name },
        subtitleProperty: { kind: "null", value: null },
        imageProperty: { kind: "null", value: null },
      };
    })
    .filter((item) => item !== null);

  return {
    items,
    details: {
      totalItems,
      nextPage: currentPage * pageSize < totalItems ? currentPage + 1 : null,
    },
  };
});

driver("details", async function (context) {
  const { z } = await import("npm:zod");

  const { externalId } = z.object({
    externalId: z.string().trim().min(1, "externalId is required"),
  }).parse(context ?? {});

  const artist = await mbGet(`artist/${externalId}`, {
    inc: "aliases",
  });

  if (!artist) {
    throw new Error(`MusicBrainz artist not found: ${externalId}`);
  }

  const name = getString(artist?.name);
  if (!name) {
    throw new Error("MusicBrainz artist is missing name");
  }

  const lifeSpan = artist?.["life-span"];
  const birthDate =
    lifeSpan && typeof lifeSpan === "object"
      ? getNullableString(lifeSpan.begin)
      : null;
  const deathDate =
    lifeSpan && typeof lifeSpan === "object"
      ? getNullableString(lifeSpan.end)
      : null;

  const beginArea = artist?.["begin-area"];
  const area = artist?.area;
  const birthPlace =
    getNullableString(beginArea?.name) || getNullableString(area?.name);

  const description = buildArtistDescription(artist);

  const aliases = Array.isArray(artist?.aliases) ? artist.aliases : [];
  const alternateNames = aliases
    .map((alias) => getNullableString(alias?.name))
    .filter((n) => n !== null && n !== name);

  return {
    name,
    properties: {
      birthDate,
      deathDate,
      birthPlace,
      description,
      alternateNames,
      assets: { images: [], videos: [] },
      sourceUrl: `https://musicbrainz.org/artist/${externalId}`,
    },
  };
});
