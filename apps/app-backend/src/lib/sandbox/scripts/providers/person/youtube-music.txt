function parseJsonResponse(responseBody) {
  try {
    return JSON.parse(responseBody);
  } catch {
    throw new Error("YouTube Music returned invalid JSON");
  }
}

function getString(value) {
  return typeof value === "string" ? value.trim() : "";
}

function getThumbnailUrls(thumbnails) {
  if (!Array.isArray(thumbnails) || thumbnails.length === 0) {
    return [];
  }

  return [...thumbnails]
    .sort(
      (a, b) =>
        (b.width || 0) * (b.height || 0) - (a.width || 0) * (a.height || 0),
    )
    .map((t) => t.url)
    .filter((url) => typeof url === "string" && url);
}

function getLargestThumbnail(thumbnails) {
  return getThumbnailUrls(thumbnails)[0] || null;
}

const YT_MUSIC_KEY = "AIzaSyC9XL3ZjWddXya6X74dJoCTL-WEYFDNX30";
const YT_MUSIC_BASE = "https://music.youtube.com/youtubei/v1";
// Protobuf-encoded filter: artists only
const ARTISTS_PARAMS = "EgWKAQIgAWoKEAkQBRAKEAMQBA==";

const YT_CONTEXT = {
  client: {
    hl: "en",
    gl: "US",
    clientName: "WEB_REMIX",
    clientVersion: "1.20231201.01.00",
  },
};

async function ytPost(endpoint, body) {
  const url = `${YT_MUSIC_BASE}/${endpoint}?key=${YT_MUSIC_KEY}&prettyPrint=false`;
  const response = await httpCall("POST", url, {
    body: JSON.stringify({ context: YT_CONTEXT, ...body }),
    headers: {
      Accept: "application/json",
      Origin: "https://music.youtube.com",
      "Content-Type": "application/json",
      "User-Agent": "Mozilla/5.0 (compatible; Ryot/1.0)",
      "X-YouTube-Client-Name": "67",
      "X-YouTube-Client-Version": "1.20231201.01.00",
    },
  });

  if (!response?.success) {
    throw new Error(
      response?.error ?? `YouTube Music request failed: ${endpoint}`,
    );
  }

  return parseJsonResponse(response.data.body);
}

function extractSectionContents(data) {
  const tabs = data?.contents?.tabbedSearchResultsRenderer?.tabs;
  if (Array.isArray(tabs)) {
    return (
      tabs[0]?.tabRenderer?.content?.sectionListRenderer?.contents ?? []
    );
  }
  return data?.contents?.sectionListRenderer?.contents ?? [];
}

driver("search", async function (context) {
  const query =
    typeof context?.query === "string" ? context.query.trim() : "";
  if (!query) {
    throw new Error("query is required");
  }

  const data = await ytPost("search", { query, params: ARTISTS_PARAMS });
  const sections = extractSectionContents(data);

  const items = [];
  for (const section of sections) {
    const shelf = section?.musicShelfRenderer;
    if (!shelf) {
      continue;
    }

    for (const item of shelf.contents ?? []) {
      const r = item?.musicResponsiveListItemRenderer;
      if (!r) {
        continue;
      }

      // Artist identifier is in the navigation endpoint's browseId (UC...)
      const browseId = getString(
        r.navigationEndpoint?.browseEndpoint?.browseId,
      );
      if (!browseId || !browseId.startsWith("UC")) {
        continue;
      }

      const name =
        getString(
          r.flexColumns?.[0]?.musicResponsiveListItemFlexColumnRenderer?.text
            ?.runs?.[0]?.text,
        ) || browseId;
      const thumbnails =
        r.thumbnail?.musicThumbnailRenderer?.thumbnail?.thumbnails ?? [];
      const imageUrl = getLargestThumbnail(thumbnails);

      items.push({
        identifier: browseId,
        badgeProperty: { kind: "null", value: null },
        titleProperty: { kind: "text", value: name },
        subtitleProperty: { kind: "null", value: null },
        imageProperty:
          imageUrl === null
            ? { kind: "null", value: null }
            : { kind: "image", value: { kind: "remote", url: imageUrl } },
      });
    }
  }

  return {
    items,
    details: { totalItems: 100, nextPage: null },
  };
});

driver("details", async function (context) {
  const identifier =
    typeof context?.identifier === "string"
      ? context.identifier.trim()
      : "";
  if (!identifier) {
    throw new Error("identifier is required");
  }

  const data = await ytPost("browse", { browseId: identifier });

  // Artist name can appear in several header renderer variants
  const header = data?.header;
  const name =
    getString(
      header?.musicImmersiveHeaderRenderer?.title?.runs?.[0]?.text,
    ) ||
    getString(
      header?.musicVisualHeaderRenderer?.title?.runs?.[0]?.text,
    ) ||
    getString(header?.musicHeaderRenderer?.title?.runs?.[0]?.text);

  if (!name) {
    throw new Error("YouTube Music artist is missing name");
  }

  // Prefer the foreground/thumbnail from whichever header variant is present
  const headerThumbnails =
    header?.musicImmersiveHeaderRenderer?.thumbnail?.musicThumbnailRenderer
      ?.thumbnail?.thumbnails ??
    header?.musicImmersiveHeaderRenderer?.foregroundThumbnail
      ?.musicThumbnailRenderer?.thumbnail?.thumbnails ??
    header?.musicVisualHeaderRenderer?.foregroundThumbnail
      ?.musicThumbnailRenderer?.thumbnail?.thumbnails ??
    header?.musicHeaderRenderer?.thumbnail?.musicThumbnailRenderer?.thumbnail
      ?.thumbnails ??
    [];

  // Description may be in the immersive header
  const descriptionRuns =
    header?.musicImmersiveHeaderRenderer?.description?.runs ?? [];
  const description =
    descriptionRuns.length > 0
      ? descriptionRuns.map((r) => getString(r.text)).join("") || null
      : null;

  return {
    name,
    properties: {
      gender: null,
      website: null,
      birthDate: null,
      deathDate: null,
      birthPlace: null,
      description,
      alternateNames: [],
      sourceUrl: `https://music.youtube.com/channel/${identifier}`,
      assets: { remoteImages: getThumbnailUrls(headerThumbnails) },
    },
  };
});
