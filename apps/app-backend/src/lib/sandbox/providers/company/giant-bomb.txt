function parseJsonResponse(responseBody) {
  try {
    return JSON.parse(responseBody);
  } catch {
    throw new Error("GiantBomb returned invalid JSON");
  }
}

async function getApiKey() {
  const configValueResponse = await getAppConfigValue(
    "videoGames.giantBomb.apiKey",
  );
  if (!configValueResponse?.success) {
    throw new Error(
      configValueResponse?.error ?? "Failed to retrieve GiantBomb API key",
    );
  }

  const apiKey =
    typeof configValueResponse.data === "string"
      ? configValueResponse.data.trim()
      : null;

  if (!apiKey) {
    throw new Error(
      "GiantBomb API key is not configured. Set VIDEO_GAMES_GIANT_BOMB_API_KEY in your environment.",
    );
  }

  return apiKey;
}

function getPrioritizedImage(image) {
  if (!image || typeof image !== "object") {
    return null;
  }

  const candidates = [
    image.original_url,
    image.super_url,
    image.medium_url,
    image.screen_large_url,
    image.screen_url,
    image.small_url,
    image.thumb_url,
    image.icon_url,
    image.tiny_url,
  ];

  for (const candidate of candidates) {
    if (typeof candidate === "string" && candidate.trim()) {
      return candidate.trim();
    }
  }

  return null;
}

function extractYear(dateStr) {
  if (typeof dateStr !== "string" || !dateStr.trim()) {
    return null;
  }
  const match = dateStr.match(/^(\d{4})/);
  if (!match) {
    return null;
  }
  const year = Number.parseInt(match[1], 10);
  return Number.isFinite(year) ? year : null;
}

function combineDescription(deck, description) {
  const deckTrimmed = typeof deck === "string" ? deck.trim() : null;
  const descTrimmed =
    typeof description === "string" ? description.trim() : null;

  if (deckTrimmed && descTrimmed) {
    return `${deckTrimmed}\n\n${descTrimmed}`;
  }

  return deckTrimmed ?? descTrimmed ?? null;
}

const BASE_URL = "https://www.giantbomb.com/api";
const GUID_PATTERN = /^\d+-\d+$/;

driver("search", async function(context) {
  const pageSize =
    Number.isFinite(Number(context?.pageSize)) &&
    Number(context.pageSize) > 0 &&
    Number(context.pageSize) <= 100
      ? Math.floor(Number(context.pageSize))
      : 20;
  const currentPage = Number.isFinite(Number(context?.page))
    ? Math.max(1, Math.floor(Number(context.page)))
    : 1;
  const query =
    typeof context?.query === "string" ? context.query.trim() : "";

  if (!query) {
    throw new Error("query is required");
  }

  const apiKey = await getApiKey();
  const offset = (currentPage - 1) * pageSize;

  const url = `${BASE_URL}/search/?api_key=${encodeURIComponent(apiKey)}&format=json&query=${encodeURIComponent(query)}&resources=company&limit=${pageSize}&offset=${offset}`;

  const response = await httpCall("GET", url, {
    headers: { Accept: "application/json" },
  });

  if (!response?.success) {
    throw new Error(response?.error ?? "GiantBomb company search request failed");
  }

  const payload = parseJsonResponse(response.data.body);

  if (payload?.error && payload.error !== "OK") {
    throw new Error(`GiantBomb API error: ${payload.error}`);
  }

  const totalItems =
    typeof payload?.number_of_total_results === "number" &&
    Number.isFinite(payload.number_of_total_results)
      ? Math.max(0, Math.trunc(payload.number_of_total_results))
      : 0;

  const results = Array.isArray(payload?.results) ? payload.results : [];

  const items = results
    .map((company) => {
      if (!company || typeof company !== "object") {
        return null;
      }

      const externalId =
        typeof company.guid === "string" && company.guid.trim()
          ? company.guid.trim()
          : null;
      if (!externalId) {
        return null;
      }

      const name =
        typeof company.name === "string" && company.name.trim()
          ? company.name.trim()
          : null;
      if (!name) {
        return null;
      }

      const image = getPrioritizedImage(company.image);

      return {
        externalId,
        calloutProperty: { kind: "null", value: null },
        titleProperty: { kind: "text", value: name },
        primarySubtitleProperty: { kind: "null", value: null },
        secondarySubtitleProperty: { kind: "null", value: null },
        imageProperty:
          image === null
            ? { kind: "null", value: null }
            : { kind: "image", value: { type: "remote", url: image } },
      };
    })
    .filter((item) => item !== null);

  return {
    items,
    details: {
      totalItems,
      nextPage: currentPage * pageSize < totalItems ? currentPage + 1 : null,
    },
  };
});

driver("details", async function(context) {
  const contextIdentifier =
    typeof context?.externalId === "string"
      ? context.externalId.trim()
      : "";

  if (!contextIdentifier) {
    throw new Error("externalId is required");
  }

  if (!GUID_PATTERN.test(contextIdentifier)) {
    throw new Error(
      "externalId must be a GiantBomb GUID (e.g., '4010-1')",
    );
  }

  const apiKey = await getApiKey();

  const fieldList = [
    "id",
    "guid",
    "name",
    "deck",
    "description",
    "date_founded",
    "location_city",
    "location_state",
    "location_country",
    "website",
    "image",
    "site_detail_url",
    "aliases",
  ].join(",");

  const url = `${BASE_URL}/company/${encodeURIComponent(contextIdentifier)}/?api_key=${encodeURIComponent(apiKey)}&format=json&field_list=${fieldList}`;

  const response = await httpCall("GET", url, {
    headers: { Accept: "application/json" },
  });

  if (!response?.success) {
    throw new Error(response?.error ?? "GiantBomb company details request failed");
  }

  const payload = parseJsonResponse(response.data.body);

  if (payload?.error && payload.error !== "OK") {
    throw new Error(`GiantBomb API error: ${payload.error}`);
  }

  const company =
    payload?.results && typeof payload.results === "object"
      ? payload.results
      : null;

  if (!company) {
    throw new Error("GiantBomb returned no company data");
  }

  const name =
    typeof company.name === "string" && company.name.trim()
      ? company.name.trim()
      : null;

  if (!name) {
    throw new Error("GiantBomb company payload is missing name");
  }

  const images = [];
  const primaryImage = getPrioritizedImage(company.image);
  if (primaryImage) {
    images.push({ type: "remote", url: primaryImage });
  }

  const locationParts = [
    typeof company.location_city === "string" ? company.location_city.trim() : null,
    typeof company.location_state === "string" ? company.location_state.trim() : null,
    typeof company.location_country === "string" ? company.location_country.trim() : null,
  ].filter((p) => p !== null && p.length > 0);
  const headquarters = locationParts.length > 0 ? locationParts.join(", ") : null;

  const foundedYear = extractYear(company.date_founded);

  const alternateNames =
    typeof company.aliases === "string" && company.aliases.trim()
      ? company.aliases
          .split("\n")
          .map((a) => a.trim())
          .filter((a) => a.length > 0)
      : [];

  const website =
    typeof company.website === "string" && company.website.trim()
      ? company.website.trim()
      : null;

  return {
    name,
    properties: {
      images,
      website,
      foundedYear,
      headquarters,
      alternateNames,
      description: combineDescription(company.deck, company.description),
      sourceUrl:
        typeof company.site_detail_url === "string" &&
        company.site_detail_url.trim()
          ? company.site_detail_url.trim()
          : null,
    },
  };
});
